# --- LOOK AT THE CUMULATIVE VARIANCE EXPLAINED AS A FUNCTION OF THE # OF PRINCIPLE COMPONENTS USED ---

# compute the fraction of variance explained by each principle component
pc_fraction_var_explained = np.cumsum(pca.explained_variance_) / np.sum(pca.explained_variance_)

# viz
plt.clf()
plt.scatter(np.arange(0,pc_fraction_var_explained.size) + 1, pc_fraction_var_explained)
plt.xlabel("Number of principle components")
plt.ylabel("Fraction of variance explained")
plt.title("Cumulative variance explained by top # of principle components")
plt.show()
